function initPlayer(x,y) { player = game.add.sprite(x, y, 'dude'); } function updatePlayer() { movePlayer(); } function movePlayer() { cursors = game.input.keyboard.createCursorKeys(); if (cursors.left.isDown) { player.body.velocity.x = -150; } else if (cursors.right.isDown) { player.body.velocity.x = 150; } else if (cursors.up.isDown) { player.body.velocity.y = 150; } else if (cursors.down.isDown) { player.body.velocity.y = -150; } }
